```javascript
/**
 * 取所有文件表头的并集，按出现顺序排列
 * @param {string[][]} fileHeadersArray - 每个文件的表头数组
 * @returns {string[]} 统一后的表头数组
 */
function computeUnifiedHeaders(fileHeadersArray) {
  var seen = {};
  var result = [];
  for (var f = 0; f < fileHeadersArray.length; f++) {
    var headers = fileHeadersArray[f];
    for (var c = 0; c < headers.length; c++) {
      var col = headers[c];
      if (!seen[col]) {
        seen[col] = true;
        result.push(col);
      }
    }
  }
  return result;
}

/**
 * 将一行数据按统一表头对齐，缺失列留空
 * @param {string[]} row - 原始行
 * @param {Object} mapping - { colName: index } 当前文件的列名→索引映射
 * @param {string[]} unifiedHeaders - 统一表头
 * @returns {string[]} 对齐后的行
 */
function alignRow(row, mapping, unifiedHeaders) {
  var result = [];
  for (var i = 0; i < unifiedHeaders.length; i++) {
    var colName = unifiedHeaders[i];
    if (mapping[colName] !== undefined) {
      result.push(row[mapping[colName]] || "");
    } else {
      result.push("");
    }
  }
  return result;
}

/**
 * 对齐多个文件的数据列
 * @param {Array<{headers: string[], data: string[][]}>} filesData
 * @returns {{ headers: string[], allData: string[][] }}
 */
function alignColumns(filesData) {
  var fileHeadersArray = filesData.map(function (f) { return f.headers; });
  var unifiedHeaders = computeUnifiedHeaders(fileHeadersArray);

  var allData = [];
  for (var fi = 0; fi < filesData.length; fi++) {
    var file = filesData[fi];
    var mapping = {};
    for (var ci = 0; ci < file.headers.length; ci++) {
      mapping[file.headers[ci]] = ci;
    }
    for (var ri = 0; ri < file.data.length; ri++) {
      allData.push(alignRow(file.data[ri], mapping, unifiedHeaders));
    }
  }
  return { headers: unifiedHeaders, data: allData };
}

/**
 * 序列化一行 CSV 字段（处理引号和分隔符）
 * @param {string[]} row
 * @param {string} delimiter
 * @returns {string}
 */
function serializeRow(row, delimiter) {
  var sep = delimiter || ",";
  var fields = [];
  for (var i = 0; i < row.length; i++) {
    var field = row[i];
    if (field === null || field === undefined) {
      field = "";
    } else {
      field = String(field);
    }
    if (field.indexOf(sep) !== -1 || field.indexOf('"') !== -1 || field.indexOf("\n") !== -1 || field.indexOf("\r") !== -1) {
      field = '"' + field.replace(/"/g, '""') + '"';
    }
    fields.push(field);
  }
  return fields.join(sep);
}

/**
 * 生成 CSV Blob 并触发下载
 * @param {string[]} headers
 * @param {string[][]} data
 * @param {string} encoding - "utf-8" | "gbk" | etc.
 * @param {string} delimiter
 * @param {string} filename
 */
function buildCSVBlob(headers, data, encoding, delimiter, filename) {
  var sep = delimiter || ",";
  var lines = [];

  if (encoding === "utf-8" || encoding === "utf8") {
    lines.push("﻿");
  }

  lines.push(serializeRow(headers, sep));
  for (var i = 0; i < data.length; i++) {
    lines.push(serializeRow(data[i], sep));
  }

  var mimeType = encoding === "utf-8" || encoding === "utf8"
    ? "text/csv;charset=utf-8"
    : "text/csv;charset=" + encoding;

  var blob = new Blob(lines, { type: mimeType });
  var url = URL.createObjectURL(blob);
  var a = document.createElement("a");
  a.href = url;
  a.download = filename || "merged.csv";
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);
}

module.exports = {
  computeUnifiedHeaders: computeUnifiedHeaders,
  alignColumns: alignColumns,
  serializeRow: serializeRow,
  buildCSVBlob: buildCSVBlob,
};
```

## Task 4: 创建 csv-merge-taskpane.js

**文件：** 创建: `src/taskpane/csv-merge-taskpane.js`

关键逻辑：
1. 文件选择 → 存储到 `selectedFiles` 数组
2. 每个文件独立检测编码/分隔符（复用 csvUtils 的读取能力）
3. 按表头模式处理数据
4. 调用 `csvMergeUtils.alignColumns` 对齐列
5. 调用 `csvMergeUtils.buildCSVBlob` 触发下载

注意：`csv-merge-utils.js` 需要新增 `readCSVFileFull` 函数用于全量读取小文件。

- [ ] **Step 1: 写实现**

```javascript
var csvUtils = require("../utils/csv-utils");
var csvMergeUtils = require("../utils/csv-merge-utils");

var selectedFiles = []; // { file, encoding, delimiter }

// ── 文件选择 ────────────────────────────────────────────────

document.getElementById("fileInput").addEventListener("change", function (e) {
  var files = Array.prototype.slice.call(e.target.files);
  files.forEach(function (f) {
    selectedFiles.push({ file: f, encoding: "utf-8", delimiter: "," });
  });
  renderFileList();
  updateMergeBtn();
});

// ── 文件列表渲染 ────────────────────────────────────────────

function renderFileList() {
  var list = document.getElementById("fileList");
  list.innerHTML = "";
  selectedFiles.forEach(function (item, index) {
    var div = document.createElement("div");
    div.className = "file-list-item";
    div.innerHTML = '<span>' + item.file.name + ' (' + item.file.size + ' bytes)</span>' +
      '<button class="remove-btn" data-index="' + index + '">✕</button>';
    list.appendChild(div);
  });
  list.querySelectorAll(".remove-btn").forEach(function (btn) {
    btn.addEventListener("click", function () {
      var idx = parseInt(this.dataset.index, 10);
      selectedFiles.splice(idx, 1);
      renderFileList();
      updateMergeBtn();
    });
  });
}

function updateMergeBtn() {
  document.getElementById("mergeBtn").disabled = selectedFiles.length === 0;
}

// ── 合并主流程 ──────────────────────────────────────────────

document.getElementById("mergeBtn").addEventListener("click", function () {
  if (selectedFiles.length === 0) return;
  setStatus("开始合并...", "loading");
  document.getElementById("mergeBtn").disabled = true;

  var headerMode = document.querySelector("input[name='headerMode']:checked").value;
  var outputEncoding = document.getElementById("outputEncoding").value;
  var outputDelimiter = document.getElementById("outputDelimiter").value;

  var filesData = [];
  var chain = Promise.resolve();

  selectedFiles.forEach(function (item) {
    chain = chain.then(function () {
      return csvMergeUtils.readCSVFileFull(item.file, item.delimiter, item.encoding);
    }).then(function (fileResult) {
      // 处理表头模式
      if (headerMode === "skip") {
        // 跳过所有表头：没有统一表头，数据直接合并
        // 需要从第一个文件提取表头作为输出表头
        filesData.push({ headers: [], data: fileResult.data });
      } else if (headerMode === "keep-first") {
        if (filesData.length === 0) {
          // 第一个文件：保留表头
          filesData.push({ headers: fileResult.headers, data: fileResult.data });
        } else {
          // 后续文件：跳过表头
          filesData.push({ headers: [], data: fileResult.data });
        }
      } else {
        // all-data：表头也作为数据
        var allRows = [fileResult.headers].concat(fileResult.data);
        filesData.push({ headers: [], data: allRows });
      }
    });
  });

  chain.then(function () {
    // 如果是 skip 模式，从第一个有数据的文件取表头
    var aligned = csvMergeUtils.alignColumns(filesData);

    csvMergeUtils.buildCSVBlob(
      aligned.headers,
      aligned.data,
      outputEncoding,
      outputDelimiter,
      "merged.csv"
    );

    setStatus("完成! 已合并 " + selectedFiles.length + " 个文件，共 " + aligned.data.length + " 行数据", "success");
    document.getElementById("mergeBtn").disabled = false;
  }).catch(function (err) {
    setStatus("错误: " + (err && err.message ? err.message : err), "error");
    document.getElementById("mergeBtn").disabled = false;
  });
});

function setStatus(msg, type) {
  var el = document.getElementById("status");
  el.textContent = "状态：" + msg;
  el.className = "status-message status-" + (type || "");
}

// Office.onReady
Office.onReady(function () {
  // Nothing special needed for this taskpane
});
```

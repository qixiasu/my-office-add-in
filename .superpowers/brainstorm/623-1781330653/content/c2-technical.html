分页查询 + 逐批写入的循环

核心循环（伪代码）
function 导出数据(原始SQL, 工作表名) {
  var 偏移量 = 0;
  var 总行数 = 0;
  var 每批大小 = 5000;

  function 处理下一批() {
    // 1️⃣ yield 给浏览器更新 UI
    setTimeout(function() {

      // 2️⃣ 查一小批
      var sql = 'SELECT * FROM (' + 原始SQL + ') LIMIT 5000 OFFSET ' + 偏移量;
      var 结果 = dbManager.exec(sql); ← 快！只查 5000 行

      // 3️⃣ 如果没数据了 → 完成
      if (结果.行数 === 0) { 显示完成(); return; }

      // 4️⃣ 写入 Excel
      Excel.run(function(ctx) {
        /* 创建/追加工作表 */
      }).then(function() {
        总行数 += 结果.行数;
        偏移量 += 5000;
        更新进度("已写入 " + 总行数 + " 行");
        处理下一批(); // 递归到下一批
      });
    });
  }

  处理下一批();
}

为什么这样能消除冻结？

  • 每次查询只取 5000 行 — sql.js 在 WASM 中扫描 5000 行几乎瞬间完成（毫秒级）
  • 查完后才写入 Excel — Excel.run 本身是异步的，给浏览器喘息机会
  • setTimeout(0) 在每批之间 yield — 让浏览器有时间处理 UI 更新、按钮状态、进度条
  • 总时间 ≈ 一次全量查询 + 逐批写入 — 总耗时不变，但不再有长时间冻结

适用场景

  • ✅ 预览模式（原始 SQL 被自动追加了 LIMIT 200）
  • ✅ 用户显式写了 LIMIT 的查询（结果已在内存中，无需分页查询）
